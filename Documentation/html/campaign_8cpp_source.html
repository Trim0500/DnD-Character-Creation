257 void Campaign::AddMapToCampaign(const int& _row, const int& _col, Map::Map& _mapToAdd) {
258 if (mapsInCampaign.size() >= mapIDs.size() * mapIDs[0].size()) {
259 throw std::exception("Too many maps!");
260 }
261 else if ((_row < 1 || _row > (int)mapIDs.size()) || (_col < 1 || _col > mapIDs[_row - 1].size())) {
262 std::ostringstream message;
263 message << "Invalid coordiantes: " << _row << "," << _col;
264 throw std::invalid_argument(message.str());
265 }
266
267 if (mapIDs[_row - 1][_col -1] == 0) {
268 mapsInCampaign.push_back(&_mapToAdd);
269 }
270 else {
271 int mapIDAtCell = mapIDs[_row - 1][_col -1];
272 auto mapAtCell = std::find_if(mapsInCampaign.begin(), mapsInCampaign.end(), [mapIDAtCell](Map::Map* map){return map->GetMapID() == mapIDAtCell;});
273 if (mapAtCell == mapsInCampaign.end()) {
274 throw std::exception("Failed to find the map to replace through update");
275 }
276
277 mapsInCampaign.erase(mapAtCell);
278 mapsInCampaign.push_back(&_mapToAdd);
279 }
280
281 mapIDs[_row - 1][_col - 1] = _mapToAdd.GetMapID();
282 }
