159 //initial list of visited cells (none to begin with)
160 std::vector<std::vector<bool>> visited(rows, std::vector<bool>(cols, false));
161
162 //BFS queue
163 std::queue<std::pair<int, int>> q;
164
165 //start cell is visited (0,0)
166 //q.push({ 0,0 });
167 q.push({ this->startCell[0], this->startCell[1] });
168
169 while (!q.empty()) {
170 auto curr = q.front(); // current checking cell
171 q.pop();
172
173 //std::cout << curr.first << ", " << curr.second << std::endl;
174 //check if reach end cell
175 if (curr.first == this->endCell[0] && curr.second == this->endCell[1]) {
176 //if made it this far, there is a path.
177 std::cout << "There is a path." << std::endl;
178 return true;
179 }
180
181 //keep searching
182 for (const auto& dir : dirs) {
183 int nextRow = curr.first + dir.first;
184 int nextCol = curr.second + dir.second;
185
186 //std::cout << "nextrow: " << nextRow << " nextcol: " << nextCol << std::endl;
187
188 //check if next cell is not a wall or outside the map or been visited
189 if (ValidCell(nextRow, nextCol, visited)) {
190 q.push({ nextRow, nextCol });
191 visited[nextRow][nextCol] = true;

193 }

195 std::cout << "There is no path. Change your map." << std::endl;
